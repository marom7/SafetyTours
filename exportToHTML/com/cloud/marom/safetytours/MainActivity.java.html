263      void setSpinners(int id_spinner , ArrayList<GeneralData> listOfItems) { 
264          // Get reference of widgets from XML layout 
265          final Spinner spinner =  findViewById(id_spinner); 
266          final List<GeneralData> pList =listOfItems; 
267   
268          ArrayAdapter<GeneralData> dataAdapter = new ArrayAdapter<GeneralData>(this, android.R.layout.simple_spinner_item, pList) { 
269              @Override 
270              public boolean isEnabled(int position) { 
271                  if (position == 0) { 
272                      // Disable the first item from Spinner 
273                      // First item will be use for hint 
274                      return false; 
275                  } else { 
276                      return true; 
277                  } 
278              } 
279   
280              @Override 
281              public View getDropDownView(int position, View convertView, 
282                                          ViewGroup parent) { 
283                  View view = super.getDropDownView(position, convertView, parent); 
284                  TextView tv = (TextView) view; 
285                  if (position == 0) { 
286                      // Set the hint text color gray 
287                      tv.setTextColor(Color.GRAY); 
288                  } else { 
289                      tv.setTextColor(Color.BLACK); 
290                  } 
291                  return view; 
292              } 
293          }; 
294          dataAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item); 
295          spinner.setAdapter(dataAdapter); 
296   
297          spinner.setOnItemSelectedListener(new AdapterView.OnItemSelectedListener() { 
298              @Override 
299              public void onItemSelected(AdapterView<?> parent, View view, int position, long id) { 
300                  // First item will be gray 
301                  if (position == 0) 
302                      ((TextView) view).setTextColor(ContextCompat.getColor(getApplicationContext(), R.color.hint_text)); 
303                   else 
304                      ((TextView) view).setTextColor(ContextCompat.getColor(getApplicationContext(), R.color.black_text)); 
305              } 
306   
307              @Override 
308              public void onNothingSelected(AdapterView<?> parent) { 
309   
310              } 
311          }); 
312   
313      }

357      private void GetUserLogIn(final String user, String pass) { 
358          // READ data from db server. 
359          HashMap<String, String> map = new HashMap<>(); 
360   
361          map.put("Content-Type", "application/json; charset=utf-8"); 
362          map.put("empNumber", user); 
363          map.put("pass", pass); 
364   
365          final PostRequest request2 = new PostRequest(Consts.getFullUrl("/SafeTourLogIn"), map, new Response.Listener<String>() { 
366              UserLogin userLogin; 
367              @Override 
368              public void onResponse(String response) { 
369   
370                  userLogin = new Gson().fromJson(response, UserLogin.class); 
371                  if(userLogin!=null && userLogin.isLogIn()) 
372                      SetDataOnUI(userLogin.getName(), userLogin.getDuty()); 
373                  else 
374                  { 
375                      Toast.makeText(getApplicationContext(), "סיסמא לא נכונה", Toast.LENGTH_LONG).show(); 
376                      openPasswordDialog(); 
377                      // finish(); 
378                  } 
379              } 
380          }, 
381                  new Response.ErrorListener() { 
382                      @Override 
383                      public void onErrorResponse(VolleyError error) { 
384                          Toast toast = Toast.makeText(getApplicationContext(), error.getMessage(), Toast.LENGTH_LONG); 
385                          toast.show(); 
386                      } 
387                  } 
388                  ); 
389   
390          Safetyapp.getInstance().getQueue().add(request2); 
391      }
